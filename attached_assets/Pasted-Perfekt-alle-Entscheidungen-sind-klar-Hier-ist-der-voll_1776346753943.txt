```typescript
/** Summe der aktiven Kaufnebenkosten in € */
function computeTotalAcquisitionCosts(purchasePrice: number, costItems: AcquisitionCostItem[]): number {
  const totalPercent = costItems.filter(c => c.active).reduce((sum, c) => sum + c.percent, 0);
  return purchasePrice * totalPercent / 100;
}

/** Gesamtfläche */
function computeTotalArea(units: RentRollUnit[]): number {
  return units.reduce((sum, u) => sum + u.area, 0);
}

/** Jahresmiete aus Rent Roll (aktuelle/Ist-Miete) */
function computeAnnualRent(units: RentRollUnit[]): number {
  return units.reduce((sum, u) => sum + u.monthlyRent, 0) * 12;
}

/** Gewichteter Ø ERV nach Fläche */
function computeWeightedERV(units: RentRollUnit[]): number {
  const totalArea = computeTotalArea(units);
  if (totalArea === 0) return 0;
  return units.reduce((sum, u) => sum + u.ervPerSqm * u.area, 0) / totalArea;
}

/** WALT in Jahren */
function computeWALT(units: RentRollUnit[], referenceDate: string): number {
  const ref = new Date(referenceDate);
  let weightedSum = 0;
  let totalArea = 0;
  units.forEach(u => {
    if (u.leaseEnd && u.tenant) {
      const end = new Date(u.leaseEnd);
      const remainingYears = Math.max(0, (end.getTime() - ref.getTime()) / (365.25 * 86400 * 1000));
      weightedSum += u.area * remainingYears;
      totalArea += u.area;
    }
  });
  return totalArea > 0 ? weightedSum / totalArea : 0;
}

/** Summe nicht-umlagefähige NK aus Rent Roll in €/Jahr */
function computeTotalNonRecoverableFromRentRoll(units: RentRollUnit[]): number {
  return units.reduce((sum, u) => sum + u.nonRecoverableOpex, 0) * 12;
}

/** Gesamt-Baubudget inkl. Contingency */
function computeTotalDevBudget(gewerke: GewerkePosition[], contingencyPercent: number): number {
  const base = gewerke.reduce((sum, g) => sum + g.budgetTotal, 0);
  return base * (1 + contingencyPercent / 100);
}

/** Gesamtkapitalbedarf */
function computeTotalCapitalRequirement(pd: PropertyData): number {
  const acqCosts = computeTotalAcquisitionCosts(pd.purchasePrice, pd.acquisitionCosts);
  const devBudget = pd.dealType === 'Development'
    ? computeTotalDevBudget(pd.gewerke, pd.contingencyPercent)
    : 0;
  return pd.purchasePrice + acqCosts + devBudget;
}

/** Gesamtdarlehen */
function computeTotalLoan(tranches: FinancingTranche[]): number {
  return tranches.reduce((sum, t) => sum + t.loanAmount, 0);
}

/** Eigenkapital */
function computeEquity(pd: PropertyData): number {
  return computeTotalCapitalRequirement(pd) - computeTotalLoan(pd.financingTranches);
}
```

### 2.2 Annuitätsberechnung

```typescript
/** 
 * Annuität pro Jahr bei annuitätischer Tilgung.
 * Formel: A = D × (i × (1+i)^n) / ((1+i)^n − 1)
 * wobei i = interestRate/100, n = loanTerm, D = loanAmount
 */
function computeAnnuity(loanAmount: number, interestRate: number, loanTerm: number): number {
  if (interestRate === 0) return loanAmount / loanTerm;
  const i = interestRate / 100;
  return loanAmount * (i * Math.pow(1 + i, loanTerm)) / (Math.pow(1 + i, loanTerm) - 1);
}

/**
 * Auto-Berechnung der Tilgungsrate bei Annuität im ersten Jahr.
 * Tilgungsrate = (Annuität − Zins Jahr 1) / Darlehensbetrag × 100
 */
function computeInitialAmortizationRate(tranche: FinancingTranche): number {
  if (tranche.repaymentType === 'Endfällig') return 0;
  const annuity = computeAnnuity(tranche.loanAmount, tranche.interestRate, tranche.loanTerm);
  const interestYear1 = tranche.loanAmount * tranche.interestRate / 100;
  const amortYear1 = annuity - interestYear1;
  return tranche.loanAmount > 0 ? (amortYear1 / tranche.loanAmount) * 100 : 0;
}
```

### 2.3 NOI pro Einheit (für Exit Value)

```typescript
interface UnitNOIBreakdown {
  unitId: string;
  usageType: string;
  area: number;
  annualRent: number;              // aus currentRent ODER ERV (je nach Modus)
  vacancyLoss: number;
  managementCost: number;          // proportional nach Fläche
  maintenance: number;             // per sqm × area
  insuranceAllocation: number;     // proportional nach Fläche
  propertyTaxAllocation: number;   // proportional nach Fläche
  otherOpexAllocation: number;     // proportional nach Fläche
  nonRecoverableOpex: number;      // individuell aus Unit
  otherIncomeAllocation: number;   // proportional nach Fläche
  noi: number;
}

/**
 * Berechnet NOI pro Einheit. Opex-Posten die auf Property-Ebene angegeben sind
 * (Management, Insurance, PropertyTax, OtherOpex, OtherIncome) werden proportional
 * nach Flächenanteil auf jede Einheit verteilt.
 * Maintenance ist per sqm → direkt auf Einheit.
 * NonRecoverableOpex ist individuell pro Einheit.
 */
function computeUnitNOIs(pd: PropertyData, useTargetRents: boolean): UnitNOIBreakdown[] {
  const units = useTargetRents ? pd.unitsTarget : pd.unitsAsIs;
  const totalArea = computeTotalArea(units);
  if (totalArea === 0) return [];

  return units.map(u => {
    const annualRent = u.monthlyRent * 12;
    const areaShare = u.area / totalArea;
    const grossRentAll = computeAnnualRent(units);
    const vacancyLoss = annualRent * pd.operatingCosts.vacancyRatePercent / 100;
    const mgmt = grossRentAll * pd.operatingCosts.managementCostPercent / 100 * areaShare;
    const maintenance = u.area * pd.operatingCosts.maintenanceReservePerSqm;
    const insAlloc = pd.operatingCosts.insurancePerYear * areaShare;
    const taxAlloc = pd.operatingCosts.propertyTaxPerYear * areaShare;
    const otherOpexAlloc = pd.operatingCosts.otherOpexPerYear * areaShare;
    const nonRec = u.nonRecoverableOpex * 12;
    const otherIncAlloc = pd.operatingCosts.otherIncomePerYear * areaShare;
    const noi = annualRent - vacancyLoss + otherIncAlloc
              - mgmt - maintenance - insAlloc - taxAlloc - otherOpexAlloc - nonRec;

    return {
      unitId: u.id, usageType: u.usageType, area: u.area,
      annualRent, vacancyLoss, managementCost: mgmt, maintenance,
      insuranceAllocation: insAlloc, propertyTaxAllocation: taxAlloc,
      otherOpexAllocation: otherOpexAlloc, nonRecoverableOpex: nonRec,
      otherIncomeAllocation: otherIncAlloc, noi,
    };
  });
}

/** Gesamt-NOI (Summe aller Unit-NOIs) */
function computePropertyNOI(pd: PropertyData, useTargetRents: boolean): number {
  return computeUnitNOIs(pd, useTargetRents).reduce((sum, u) => sum + u.noi, 0);
}
```

### 2.4 Exit Value pro Nutzungsart

```typescript
/**
 * Exit Value = Summe pro Einheit: Unit-NOI (im Exit-Jahr, indexiert) / Exit Cap Rate der Nutzungsart
 * - Bei Development: Mieten erst ab Fertigstellung, dann mit ervGrowth indexiert
 * - Bei Investment: Mieten ab Jahr 1, mit ervGrowth indexiert
 * - Opex werden mit opexInflation indexiert
 */
function computeExitValue(
  pd: PropertyData,
  exitYearIndex: number,        // 0-basiert
  devCompletionYearIndex: number  // 0 bei Investment, sonst Jahr der Fertigstellung
): number {
  const useTarget = pd.unitsTarget.length > 0;
  const unitNOIs = computeUnitNOIs(pd, useTarget);

  let totalExitValue = 0;

  unitNOIs.forEach(unitNOI => {
    const unit = (useTarget ? pd.unitsTarget : pd.unitsAsIs).find(u => u.id === unitNOI.unitId);
    if (!unit) return;

    const marketAssumption = pd.marketAssumptions.perUsageType.find(m => m.usageType === unit.usageType);
    const ervGrowth = marketAssumption?.ervGrowthRatePercent ?? 2.0;
    const exitCap = marketAssumption?.exitCapRatePercent ?? 5.0;
    const opexInflation = pd.marketAssumptions.opexInflationPercent;

    // Jahre seit Fertigstellung (Wachstum wirkt nur nach Fertigstellung)
    const yearsOfGrowth = Math.max(0, exitYearIndex - devCompletionYearIndex);

    // Indexierte Werte in Exit-Jahr
    const futureRent = unitNOI.annualRent * Math.pow(1 + ervGrowth / 100, yearsOfGrowth);
    const opexRatio = unitNOI.annualRent > 0
      ? (unitNOI.annualRent - unitNOI.noi - unitNOI.otherIncomeAllocation) / unitNOI.annualRent
      : 0.25;
    const futureOpex = (futureRent * opexRatio) * Math.pow(1 + opexInflation / 100, yearsOfGrowth) / Math.pow(1 + ervGrowth / 100, yearsOfGrowth);
    // Vereinfacht: futureNOI = futureRent − futureOpex
    const futureNOI = futureRent - futureOpex;

    const teilwert = futureNOI / (exitCap / 100);
    totalExitValue += teilwert;
  });

  return totalExitValue;
}
```

```typescript
interface PropertyCashFlowMonth {
  monthIndex: number;            // 0 = Ankaufsmonat

  // Betrieblich
  grossRentalIncome: number;

  // Transaktionen
  acquisitionPrice: number;
  acquisitionCosts: number;
  salesProceeds: number;
  salesCosts: number;
  transactionsCashflow: number;
  // Finanzierung
  loanReceived: number;

  // Gesamt

  cumulativeFreeCashflow: number;
}

/**
 * KERN-BERECHNUNG — monatliche Cashflows über die Haltedauer.
 * Jährliche Ansicht wird daraus aggregiert.
 */
function computePropertyCashFlowMonthly(pd: PropertyData, sale?: SaleObject): PropertyCashFlowMonth[] {
  const months: PropertyCashFlowMonth[] = [];
  const holdingYears = pd.holdingPeriodYears || 10;

  // Sale überschreibt Haltedauer
  const acqDate = new Date(pd.acquisitionDate || new Date().toISOString());
  let exitDate = new Date(acqDate);
  exitDate.setFullYear(exitDate.getFullYear() + holdingYears);

  if (sale && sale.soldAt) {
    exitDate = new Date(sale.soldAt);
  } else if (sale && (sale as any).targetCloseDate) {
    exitDate = new Date((sale as any).targetCloseDate);
  }

  const totalMonths = Math.max(1, Math.round((exitDate.getTime() - acqDate.getTime()) / (30.44 * 86400 * 1000)));

  const isDev = pd.dealType === 'Development';
  const devStartDate = isDev && pd.projectStart ? new Date(pd.projectStart) : acqDate;
  const lastGewerkEndWeek = pd.gewerke.length > 0
    ? Math.max(...pd.gewerke.map(g => g.startWeek + g.durationWeeks - 1))
    : 0;
  const devEndDate = new Date(devStartDate);
  devEndDate.setDate(devEndDate.getDate() + lastGewerkEndWeek * 7);

  // Monatsindex der Fertigstellung
  const devCompletionMonth = isDev
    ? Math.max(0, Math.round((devEndDate.getTime() - acqDate.getTime()) / (30.44 * 86400 * 1000)))
    : 0;

  // === CAPEX pro Monat berechnen ===
  const capexPerMonth = new Array(totalMonths + 2).fill(0);
  if (isDev) {
    pd.gewerke.forEach(g => {
      const gStartDate = new Date(devStartDate);
      gStartDate.setDate(gStartDate.getDate() + (g.startWeek - 1) * 7);
      const gEndDate = new Date(gStartDate);
      gEndDate.setDate(gEndDate.getDate() + g.durationWeeks * 7);
      const gStartMonth = Math.round((gStartDate.getTime() - acqDate.getTime()) / (30.44 * 86400 * 1000));
      const gEndMonth = Math.round((gEndDate.getTime() - acqDate.getTime()) / (30.44 * 86400 * 1000));
      const gDurationMonths = Math.max(1, gEndMonth - gStartMonth);

      // Kostenverteilung
      switch (g.costDistribution) {
        case 'vorauszahlung': {
          if (gStartMonth >= 0 && gStartMonth < capexPerMonth.length) {
            capexPerMonth[gStartMonth] += g.budgetTotal;
          }
          break;
        }
        case 'endfällig': {
          const endIdx = Math.min(gEndMonth - 1, capexPerMonth.length - 1);
          if (endIdx >= 0) {
            capexPerMonth[endIdx] += g.budgetTotal;
          }
          break;
        }
        case '30-40-30': {
          const firstIdx = gStartMonth;
          const midIdx = gStartMonth + Math.floor(gDurationMonths / 2);
          const lastIdx = Math.min(gEndMonth - 1, capexPerMonth.length - 1);
          if (firstIdx >= 0 && firstIdx < capexPerMonth.length) capexPerMonth[firstIdx] += g.budgetTotal * 0.30;
          if (midIdx >= 0 && midIdx < capexPerMonth.length) capexPerMonth[midIdx] += g.budgetTotal * 0.40;
          if (lastIdx >= 0 && lastIdx < capexPerMonth.length) capexPerMonth[lastIdx] += g.budgetTotal * 0.30;
          break;
        }
        case 'linear':
        default: {
          const perMonth = g.budgetTotal / gDurationMonths;
          for (let m = gStartMonth; m < gEndMonth && m < capexPerMonth.length; m++) {
            if (m >= 0) capexPerMonth[m] += perMonth;
          }
          break;
        }
      }
    });

    // Contingency proportional auf alle Capex-Monate verteilen
    const contMultiplier = 1 + pd.contingencyPercent / 100;
    for (let i = 0; i < capexPerMonth.length; i++) {
      capexPerMonth[i] *= contMultiplier;
    }

    // Capex-Inflation wirkt ab Ankaufsmonat (Monat 0 = keine Inflation,
    // Monat 12 = (1+capexInflation)^1, etc.)
    for (let i = 0; i < capexPerMonth.length; i++) {
      const yearsFromAcq = i / 12;
      capexPerMonth[i] *= Math.pow(1 + pd.marketAssumptions.capexInflationPercent / 100, yearsFromAcq);
    }
  }

  // === TRANSAKTIONEN ===
  const totalAcqCosts = computeTotalAcquisitionCosts(pd.purchasePrice, pd.acquisitionCosts);
  const exitValue = sale && sale.soldPrice
    ? sale.soldPrice
    : computeExitValue(pd, totalMonths / 12, devCompletionMonth / 12);
  const salesCostAmount = exitValue * pd.marketAssumptions.salesCostPercent / 100;

  // === FINANZIERUNG ===
  // Restschuld-Tracking pro Tranche (monatlich)
  const trancheState = pd.financingTranches.map(t => {
    const amortRate = t.repaymentType === 'Annuität'
      ? computeInitialAmortizationRate(t)
      : 0;
    return {
      ...t,
      amortizationRate: amortRate,
      outstanding: t.loanAmount,
      annuityAnnual: t.repaymentType === 'Annuität'
        ? computeAnnuity(t.loanAmount, t.interestRate, t.loanTerm)
        : 0,
    };
  });

  let cumulativeCF = 0;

  for (let m = 0; m < totalMonths; m++) {
    const calDate = new Date(acqDate);
    calDate.setMonth(calDate.getMonth() + m);
    const calYMD = `${calDate.getFullYear()}-${String(calDate.getMonth() + 1).padStart(2, '0')}`;

    // === MIETEINNAHMEN ===
    let grossRent = 0;

    // Während Bau: Ist-Mieten (nur wenn Mieter noch läuft)
    if (isDev && m < devCompletionMonth) {
      pd.unitsAsIs.forEach(u => {
        if (!u.tenant || !u.leaseStart) return;
        const leaseStart = new Date(u.leaseStart);
        const leaseEnd = u.leaseEnd ? new Date(u.leaseEnd) : null;
        if (calDate >= leaseStart && (!leaseEnd || calDate <= leaseEnd)) {
          // Indexierung: Anzahl kompletter Indexierungsintervalle seit Mietbeginn
          const yearsSinceStart = (calDate.getTime() - leaseStart.getTime()) / (365.25 * 86400 * 1000);
          let intervalYears = 0;
          switch (u.indexationInterval) {
            case 'jährlich': intervalYears = 1; break;
            case 'alle 2 Jahre': intervalYears = 2; break;
            case 'alle 3 Jahre': intervalYears = 3; break;
            case 'keine': intervalYears = 0; break;
          }
          const completedIntervals = intervalYears > 0 ? Math.floor(yearsSinceStart / intervalYears) : 0;
          const indexFactor = Math.pow(1 + u.indexationRate / 100, completedIntervals);
          grossRent += u.monthlyRent * indexFactor;
        }
      });
    } else {
      // Nach Fertigstellung (oder Investment): Ziel-Mieten ab individuellem Mietbeginn
      const targetUnits = pd.unitsTarget.length > 0 ? pd.unitsTarget : pd.unitsAsIs;
      targetUnits.forEach(u => {
        // Mietbeginn bestimmen
        let mietbeginn: Date;
        if (u.leaseStart) {
          mietbeginn = new Date(u.leaseStart);
        } else if (isDev) {
          mietbeginn = devEndDate;  // Default: Fertigstellung
        } else {
          mietbeginn = acqDate;
        }

        // Prüfe ob Mietbeginn erreicht
        if (calDate < mietbeginn) return;
        // Prüfe Mietende
        if (u.leaseEnd) {
          const leaseEnd = new Date(u.leaseEnd);
          if (calDate > leaseEnd) return;
        }

        // Indexierung mit ERV Growth der Nutzungsart (nach Fertigstellung)
        const marketAssumption = pd.marketAssumptions.perUsageType.find(ma => ma.usageType === u.usageType);
        const growth = marketAssumption?.ervGrowthRatePercent ?? 2.0;
        const yearsOfGrowth = Math.max(0, (calDate.getTime() - mietbeginn.getTime()) / (365.25 * 86400 * 1000));
        const indexFactor = Math.pow(1 + growth / 100, yearsOfGrowth);
        grossRent += u.monthlyRent * indexFactor;
      });
    }

    // === BETRIEBSKOSTEN ===
    // Basis-Opex monatlich, indexiert mit opexInflation
    const yearsFromAcq = m / 12;
    const useTarget = m >= devCompletionMonth && pd.unitsTarget.length > 0;
    const baseAnnualOpex = computeTotalArea(useTarget ? pd.unitsTarget : pd.unitsAsIs) > 0
      ? computeUnitNOIs(pd, useTarget).reduce(
          (sum, u) => sum + u.managementCost + u.maintenance + u.insuranceAllocation
                    + u.propertyTaxAllocation + u.otherOpexAllocation + u.nonRecoverableOpex, 0)
      : 0;
    const indexedAnnualOpex = baseAnnualOpex * Math.pow(1 + pd.marketAssumptions.opexInflationPercent / 100, yearsFromAcq);
    const opexMonthly = indexedAnnualOpex / 12;
    const otherIncomeMonthly = pd.operatingCosts.otherIncomePerYear * Math.pow(1 + pd.marketAssumptions.opexInflationPercent / 100, yearsFromAcq) / 12;

    // NOI
    const vacancyMonthly = grossRent * pd.operatingCosts.vacancyRatePercent / 100;
    const noi = grossRent - vacancyMonthly + otherIncomeMonthly - opexMonthly;

    // === CAPEX ===
    const capex = capexPerMonth[m] || 0;

    // === TRANSAKTIONEN ===
    const isFirstMonth = m === 0;
    const isLastMonth = m === totalMonths - 1;
    const acqPrice = isFirstMonth ? -pd.purchasePrice : 0;
    const acqCostsThisMonth = isFirstMonth ? -totalAcqCosts : 0;
    const salesProc = isLastMonth ? exitValue : 0;
    const salesCost = isLastMonth ? -salesCostAmount : 0;
    const txCF = acqPrice + acqCostsThisMonth + salesProc + salesCost;

    // === FINANZIERUNG ===
    let loanReceived = 0;
    let interestThisMonth = 0;
    let repaymentThisMonth = 0;

    trancheState.forEach(t => {
      // Auszahlung bei Disbursement Date (Default = Monat 0)
      const disbDate = t.disbursementDate ? new Date(t.disbursementDate) : acqDate;
      const disbMonth = Math.round((disbDate.getTime() - acqDate.getTime()) / (30.44 * 86400 * 1000));

      if (m === disbMonth) {
        loanReceived += t.loanAmount;
      }

      // Zinsen & Tilgung nur wenn Darlehen ausgezahlt und nicht zurückgezahlt
      if (m > disbMonth && t.outstanding > 0) {
        const monthsSinceDisb = m - disbMonth;
        const loanTermMonths = t.loanTerm * 12;

        if (monthsSinceDisb <= loanTermMonths) {
          // Monatliche Zinsen
          const monthlyInterest = t.outstanding * t.interestRate / 100 / 12;
          interestThisMonth += monthlyInterest;

          // Tilgung
          if (t.repaymentType === 'Annuität') {
            const monthlyAnnuity = t.annuityAnnual / 12;
            const monthlyRepayment = Math.max(0, monthlyAnnuity - monthlyInterest);
            const actualRepayment = Math.min(monthlyRepayment, t.outstanding);
            repaymentThisMonth += actualRepayment;
            t.outstanding -= actualRepayment;
          }
          // Bei Endfällig: keine laufende Tilgung
        }
      }

      // Im Exit-Monat: Restschuld komplett zurückzahlen
      if (isLastMonth && t.outstanding > 0) {
        repaymentThisMonth += t.outstanding;
        t.outstanding = 0;
      }
    });

    const debtCF = loanReceived - interestThisMonth - repaymentThisMonth;

    // === FREE CASHFLOW ===
    const freeCF = noi - capex + txCF + debtCF;
    cumulativeCF += freeCF;

    months.push({
      monthIndex: m,
      calendarDate: calYMD,
      grossRentalIncome: grossRent,
      operatingCosts: opexMonthly + vacancyMonthly - otherIncomeMonthly,
      capexConstructionCosts: capex,
      noi,
      acquisitionPrice: acqPrice,
      acquisitionCosts: acqCostsThisMonth,
      salesProceeds: salesProc,
      salesCosts: salesCost,
      transactionsCashflow: txCF,
      loanReceived,
      interestPayments: -interestThisMonth,
      loanRepayments: -repaymentThisMonth,
      debtCashflow: debtCF,
      freeCashflow: freeCF,
      cumulativeFreeCashflow: cumulativeCF,
    });
  }

  return months;
}
```

```typescript
interface PropertyCashFlowYear {
  yearIndex: number;
  calendarYear: number;
  grossRentalIncome: number;
  operatingCosts: number;
  capexConstructionCosts: number;
  noi: number;
  acquisitionPrice: number;
  acquisitionCosts: number;
  salesProceeds: number;
  salesCosts: number;
  transactionsCashflow: number;
  loanReceived: number;
  interestPayments: number;
  loanRepayments: number;
  debtCashflow: number;
  freeCashflow: number;
  cumulativeFreeCashflow: number;
}

function aggregateToYears(months: PropertyCashFlowMonth[], startYear: number): PropertyCashFlowYear[] {
  const years: PropertyCashFlowYear[] = [];
  const monthsPerYear: Record<number, PropertyCashFlowMonth[]> = {};

  months.forEach(m => {
    const yearIdx = Math.floor(m.monthIndex / 12);
    if (!monthsPerYear[yearIdx]) monthsPerYear[yearIdx] = [];
    monthsPerYear[yearIdx].push(m);
  });

  Object.entries(monthsPerYear).forEach(([yrStr, ms]) => {
    const yr = parseInt(yrStr);
    const agg = ms.reduce((acc, m) => ({
      grossRentalIncome: acc.grossRentalIncome + m.grossRentalIncome,
      operatingCosts: acc.operatingCosts + m.operatingCosts,
      capexConstructionCosts: acc.capexConstructionCosts + m.capexConstructionCosts,
      noi: acc.noi + m.noi,
      acquisitionPrice: acc.acquisitionPrice + m.acquisitionPrice,
      acquisitionCosts: acc.acquisitionCosts + m.acquisitionCosts,
      salesProceeds: acc.salesProceeds + m.salesProceeds,
      salesCosts: acc.salesCosts + m.salesCosts,
      transactionsCashflow: acc.transactionsCashflow + m.transactionsCashflow,
      loanReceived: acc.loanReceived + m.loanReceived,
      interestPayments: acc.interestPayments + m.interestPayments,
      loanRepayments: acc.loanRepayments + m.loanRepayments,
      debtCashflow: acc.debtCashflow + m.debtCashflow,
      freeCashflow: acc.freeCashflow + m.freeCashflow,
    }), {
      grossRentalIncome: 0, operatingCosts: 0, capexConstructionCosts: 0, noi: 0,
      acquisitionPrice: 0, acquisitionCosts: 0, salesProceeds: 0, salesCosts: 0,
      transactionsCashflow: 0, loanReceived: 0, interestPayments: 0, loanRepayments: 0,
      debtCashflow: 0, freeCashflow: 0,
    });

    const cumulative = ms[ms.length - 1].cumulativeFreeCashflow;

    years.push({
      yearIndex: yr,
      calendarYear: startYear + yr,
      ...agg,
      cumulativeFreeCashflow: cumulative,
    });
  });

  return years;
}
```

### 2.7 IRR-Berechnung (korrekt, aus Equity-Sicht)

```typescript
/**
 * Newton-Raphson IRR-Berechnung.
 * cashflows[0] muss negativ sein (Investment), Rest sind Returns.
 */
function computeIRR(cashflows: number[], guess = 0.1): number {
  const maxIter = 100;
  const tolerance = 1e-7;
  let rate = guess;

  for (let iter = 0; iter < maxIter; iter++) {
    let npv = 0;
    let dnpv = 0;
    for (let t = 0; t < cashflows.length; t++) {
      npv += cashflows[t] / Math.pow(1 + rate, t);
      if (t > 0) dnpv += -t * cashflows[t] / Math.pow(1 + rate, t + 1);
    }
    if (Math.abs(npv) < tolerance) return rate * 100;
    if (Math.abs(dnpv) < 1e-12) break;
    rate = rate - npv / dnpv;
  }
  return rate * 100;
}

/**
 * 10-Year Hold IRR (oder Haltedauer) aus Equity-Sicht.
 * Year 0: −Equity (NUR Eigenkapital, nicht Gesamtinvestition, weil Loan Received den Kaufpreis teilweise deckt)
 * Year 1 bis n-1: NOI − Debt Service (laufende Ausschüttungen)
 * Year n: NOI − Debt Service + Sales Proceeds − Sales Costs − Restschuld
 *
 * TECHNISCHE NOTE: Aus dem Monthly Cashflow können wir die EquityCashflows ableiten:
 * equityCF = freeCashflow + loanReceived − acquisitionPrice − acquisitionCosts
 * (nettiert den Acquisitions-Effekt und den Loan Received raus, weil die bereits im freeCashflow
 *  verrechnet sind. Das Ergebnis ist der Cashflow den der Equity-Investor sieht.)
 * Year 0: Der Equity-CF = Nettoeinsatz = −(Acquisitions − Loan)
 * Vereinfacht: equityCF[0] = -Equity, equityCF[n] enthält Sales Proceeds + Restschuld-Rückzahlung
 */
function computeHoldIRR(pd: PropertyData, months: PropertyCashFlowMonth[]): number {
  const years = aggregateToYears(months, new Date(pd.acquisitionDate).getFullYear());
  const equity = computeEquity(pd);

  // Erstes Jahr: Equity-Einsatz (negativ)
  // Laufende Jahre: NOI + Debt CF (der enthält bereits Zins & Tilgung negativ)
  // Capex in laufenden Jahren sind Teil der Equity-Investition (wenn nicht fremdfinanziert — vereinfacht wir annehmen dass alles EK-finanziert ist)
  const equityCashflows: number[] = [];

  years.forEach((y, idx) => {
    let cf: number;
    if (idx === 0) {
      // Year 0: Equity-Einsatz + laufender CF (NOI + Debt CF ohne Loan Received)
      const debtCFexLoan = y.debtCashflow - y.loanReceived;  // nur Zins + Tilgung
      cf = -equity + y.noi - y.capexConstructionCosts + debtCFexLoan;
    } else if (idx === years.length - 1) {
      // Letztes Jahr: NOI + Debt CF + Sales Proceeds − Sales Costs
      cf = y.noi - y.capexConstructionCosts + y.debtCashflow + y.transactionsCashflow;
    } else {
      cf = y.noi - y.capexConstructionCosts + y.debtCashflow;
    }
    equityCashflows.push(cf);
  });

  return computeIRR(equityCashflows);
}

/**
 * Development IRR — monatliche Cashflows von Ankauf bis Fertigstellung.
 * Zusätzliches letztes "Monat": Wert bei Fertigstellung (= Ziel-NOI / Exit Cap Rate) − Restschuld.
 */
function computeDevIRR(pd: PropertyData, months: PropertyCashFlowMonth[]): number | null {
  if (pd.dealType !== 'Development') return null;
  if (pd.gewerke.length === 0) return null;

  const acqDate = new Date(pd.acquisitionDate);
  const devStart = new Date(pd.projectStart);
  const lastGewerkEndWeek = Math.max(...pd.gewerke.map(g => g.startWeek + g.durationWeeks - 1));
  const devEndDate = new Date(devStart);
  devEndDate.setDate(devEndDate.getDate() + lastGewerkEndWeek * 7);

  const devEndMonth = Math.round((devEndDate.getTime() - acqDate.getTime()) / (30.44 * 86400 * 1000));

  // Cashflows von Monat 0 bis devEndMonth
  const devMonths = months.slice(0, devEndMonth + 1);
  const equity = computeEquity(pd);

  // Monatliche Equity-Cashflows
  const monthlyEquityCF: number[] = [];

  devMonths.forEach((m, idx) => {
    let cf: number;
    if (idx === 0) {
      // Equity-Einsatz + laufender CF (ohne Loan Received)
      const debtCFexLoan = m.debtCashflow - m.loanReceived;
      cf = -equity + m.noi - m.capexConstructionCosts + debtCFexLoan;
    } else {
      cf = m.noi - m.capexConstructionCosts + m.debtCashflow;
    }
    monthlyEquityCF.push(cf);
  });

  // Im letzten Monat: Wert bei Fertigstellung addieren, Restschuld abziehen
  const devEndYearIdx = devEndMonth / 12;
  const valueAtCompletion = computeExitValue(pd, devEndYearIdx, devEndYearIdx);

  // Restschuld berechnen durch Simulation
  let totalOutstanding = 0;
  pd.financingTranches.forEach(t => {
    if (t.repaymentType === 'Endfällig') {
      totalOutstanding += t.loanAmount;
    } else {
      // Annuität: Restschuld nach devEndMonth Monaten
      const monthlyAnnuity = computeAnnuity(t.loanAmount, t.interestRate, t.loanTerm) / 12;
      const monthlyRate = t.interestRate / 100 / 12;
      let outstanding = t.loanAmount;
      for (let i = 0; i < devEndMonth; i++) {
        const interest = outstanding * monthlyRate;
        const repayment = monthlyAnnuity - interest;
        outstanding = Math.max(0, outstanding - repayment);
      }
      totalOutstanding += outstanding;
    }
  });

  monthlyEquityCF[monthlyEquityCF.length - 1] += valueAtCompletion - totalOutstanding;

  // Monatliches IRR berechnen und auf Jahresbasis umrechnen
  const monthlyIRR = computeIRR(monthlyEquityCF) / 100;
  const annualIRR = (Math.pow(1 + monthlyIRR, 12) - 1) * 100;

  return annualIRR;
}
```

### 2.8 Gesamte KPIs

```typescript
interface PropertyKPIs {
  niyAtAcquisition: number;
  multiple: number;
  dscr: number;
  ltv: number;
  cashOnCashYear1: number;
  gri: number;
  noiIst: number;
  noiZiel: number;
  irr10Year: number;
  equityMultiple10Year: number;
  irrDevelopment: number | null;
  developmentProfit: number;
  profitOnCost: number;
  netDevelopmentYield: number;
  cashOnCash10YearAvg: number;
  peakEquity: number;
  paybackPeriodYears: number;
}

function computePropertyKPIs(pd: PropertyData, sale?: SaleObject): PropertyKPIs {
  const months = computePropertyCashFlowMonthly(pd, sale);
  const years = aggregateToYears(months, new Date(pd.acquisitionDate).getFullYear());
  const noiIst = computePropertyNOI(pd, false);
  const noiZiel = pd.unitsTarget.length > 0 ? computePropertyNOI(pd, true) : noiIst;

  const totalAcqCosts = computeTotalAcquisitionCosts(pd.purchasePrice, pd.acquisitionCosts);
  const totalDevBudget = pd.dealType === 'Development'
    ? computeTotalDevBudget(pd.gewerke, pd.contingencyPercent) : 0;
  const totalCostBasis = pd.purchasePrice + totalAcqCosts + totalDevBudget;
  const totalCapReq = computeTotalCapitalRequirement(pd);
  const totalLoan = computeTotalLoan(pd.financingTranches);
  const equity = totalCapReq - totalLoan;

  // Debt Service p.a. (Jahr 1)
  const totalDS = pd.financingTranches.reduce((sum, t) => {
    const interest = t.loanAmount * t.interestRate / 100;
    const amort = t.repaymentType === 'Annuität'
      ? computeAnnuity(t.loanAmount, t.interestRate, t.loanTerm) - interest
      : 0;
    return sum + interest + amort;
  }, 0);

  const irr10 = computeHoldIRR(pd, months);
  const irrDev = computeDevIRR(pd, months);

  // Equity Multiple = (Equity + Summe aller positiven Returns) / Equity
  const totalReturns = years.reduce((sum, y) => sum + Math.max(0, y.freeCashflow + (y.yearIndex === 0 ? equity : 0)), 0);
  const equityMultiple = equity > 0 ? totalReturns / equity : 0;

  // Development Profit
  let devProfit = 0;
  if (pd.dealType === 'Development' && pd.gewerke.length > 0) {
    const devStart = new Date(pd.projectStart);
    const lastWk = Math.max(...pd.gewerke.map(g => g.startWeek + g.durationWeeks - 1));
    const devEnd = new Date(devStart);
    devEnd.setDate(devEnd.getDate() + lastWk * 7);
    const devEndMonth = Math.round((devEnd.getTime() - new Date(pd.acquisitionDate).getTime()) / (30.44 * 86400 * 1000));
    const valueAtCompletion = computeExitValue(pd, devEndMonth / 12, devEndMonth / 12);

    const finCostsDuringDev = months.slice(0, devEndMonth + 1)
      .reduce((sum, m) => sum + Math.abs(m.interestPayments), 0);

    devProfit = valueAtCompletion - pd.purchasePrice - totalAcqCosts - totalDevBudget - finCostsDuringDev;
  }

  const profitOnCost = totalCostBasis > 0 ? (devProfit / totalCostBasis) * 100 : 0;
  const netDevYield = totalCostBasis > 0 ? (noiZiel / totalCostBasis) * 100 : 0;

  // Cash-on-Cash 10 Year Average
  const annualCoCs = years.slice(1).map(y => {
    const cashAfterDebt = y.noi + y.debtCashflow;
    return equity > 0 ? (cashAfterDebt / equity) * 100 : 0;
  });
  const avgCoC = annualCoCs.length > 0 ? annualCoCs.reduce((s, v) => s + v, 0) / annualCoCs.length : 0;

  // Peak Equity & Payback
  let peakEquity = 0;
  let paybackYear = years.length;
  years.forEach(y => {
    if (y.cumulativeFreeCashflow < peakEquity) peakEquity = y.cumulativeFreeCashflow;
    if (y.cumulativeFreeCashflow > 0 && y.yearIndex < paybackYear) paybackYear = y.yearIndex;
  });

  return {
    niyAtAcquisition: pd.purchasePrice > 0 ? (noiIst / pd.purchasePrice) * 100 : 0,
    multiple: noiIst > 0 ? pd.purchasePrice / (computeAnnualRent(pd.unitsAsIs)) : 0,
    dscr: totalDS > 0 ? noiZiel / totalDS : 999,
    ltv: totalCapReq > 0 ? (totalLoan / totalCapReq) * 100 : 0,
    cashOnCashYear1: equity > 0 ? ((noiIst - totalDS) / equity) * 100 : 0,
    gri: computeAnnualRent(pd.unitsTarget.length > 0 ? pd.unitsTarget : pd.unitsAsIs),
    noiIst,
    noiZiel,
    irr10Year: irr10,
    equityMultiple10Year: equityMultiple,
    irrDevelopment: irrDev,
    developmentProfit: devProfit,
    profitOnCost,
    netDevelopmentYield: netDevYield,
    cashOnCash10YearAvg: avgCoC,
    peakEquity: Math.abs(peakEquity),
    paybackPeriodYears: paybackYear,
  };
}
```

## TEIL 7: STORE-FUNKTIONEN

### Transfer-Funktionen in `src/store/useStore.ts`

```typescript
transferToDevelopment: (dealId) => {
  const deal = get().deals.find(d => d.id === dealId);
  if (!deal) return;
  
  const now = new Date().toISOString();
  const devPropertyData = JSON.parse(JSON.stringify(deal.propertyData));  // Deep copy
  const snapshot = JSON.parse(JSON.stringify(deal.propertyData));         // Deep copy für Freeze
  
  const newDev: DevelopmentProject = {
    id: `dev-${Date.now()}`,
    // ...bestehende Felder mappen aus propertyData...
    name: devPropertyData.name,
    address: devPropertyData.address,
    city: devPropertyData.city,
    // ...etc
    propertyData: devPropertyData,
    underwritingSnapshot: snapshot,
    offers: [],
    invoices: [],
    createdAt: now,
    updatedAt: now,
  };
  
  set(s => ({
    deals: s.deals.filter(d => d.id !== dealId),
    developments: [...s.developments, newDev],
  }));
},

transferDevToBestand: (devId) => {
  const dev = get().developments.find(d => d.id === devId);
  if (!dev) return;
  
  const pd = JSON.parse(JSON.stringify(dev.propertyData));  // Deep copy
  // Bei Transfer in Bestand: Ziel-Mieten werden zu Ist-Mieten
  if (pd.unitsTarget.length > 0) {
    pd.unitsAsIs = pd.unitsTarget;
    pd.unitsTarget = [];
  }
  
  const newAsset: Asset = {
    // ...bestehende Felder...
    propertyData: pd,
    // Mapping auf bestehende Asset-Felder (units, annualRent, operatingCosts, etc.)
  };
  
  // Offers und Invoices bleiben auf dem Development als historisches Dokument
  // Development wird mit Status "Abgeschlossen" markiert statt gelöscht
  
  set(s => ({
    assets: [...s.assets, newAsset],
    developments: s.developments.map(d => d.id === devId ? { ...d, status: 'Abgeschlossen' } : d),
  }));
},
```

### Auto-Save

Live Save bei jeder Änderung auf Detailseiten. `updateDeal`, `updateDevelopment`, `updateAsset` werden bei jedem Input-Blur oder Change aufgerufen. Nach erfolgreichem Save: kleine Toast-Nachricht oben rechts ("Gespeichert ✓", verschwindet nach 2 Sekunden).
